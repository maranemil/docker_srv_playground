services:
  node:
    image: "node:22"
    user: "node"
    working_dir: /home/node/app
    environment:
      - NODE_ENV=production
    volumes:
      - ./:/home/node/app
    ports: # use if it is necessary to expose the container to the host machine
      - "8888:8888"
    command: ["npm", "start"]



docker run \
  -d \
  --name node-app \
  --user node \
  --workdir /home/node/app \
  -e NODE_ENV=production \
  -v "$(pwd)":/home/node/app \
  -p 8888:8888 \
  node:22 \
  npm start
